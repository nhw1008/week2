// 피보나치 수열을 구현하자
// 피보나치 수열 결과를 반환하는 함수를
// f()라고 하자
// f() 인자로 위치를 넣으면 그 값을 반환하라

void main() {
  // f(0) -> 1
  // f(1) -> 1
  // f(2) -> 2 // f(0) + f(1)
  // f(3) -> 3 // f(1) + f(2)
  // f(4) -> 5 // f(2) + f(3)
  // f(5) -> 8 // f(3) + f(4)

  print(f(5));
}

int f(n) {
  return n < 2 ? 1 : f(n - 2) + f(n - 1);
}
